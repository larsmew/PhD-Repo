extractSamples <- function(brca_data, haplotypes_brca, distinct = F){
    "
    Extract relevant samples from table of all genotypes 
    "
    
    # Keep only one member of a family and remove the others, if chosen
    if (distinct){
        brca_data <- brca_data[!duplicated(brca_data$FamCode), ]
        # Equivalent to above
        # brca_data <- subset(brca_data, !duplicated(FamCode)) 
        # brca_data <- distinct(brca_data, FamCode, .keep_all = T)
    }
    
    # Extract samples (with given brca mutation) from the big haplotypes dataset
    # Faster to use fmatch than match
    index <- fmatch(brca_data$Onc_ID, haplotypes_brca$SNP)
    matched <- haplotypes_brca[index, ]
    
    # as above, but slower
    # matched <- filter(haplotypes_brca, SNP %in% brca_data$Onc_ID) 
    
    return(matched)
}

### Haplotypes samples with a given brca mutation ###
haplotype_mutation <- function(brca_data, haplotypes_brca, chr_coords, 
                               brca_start, brca_stop, brca_middle, brca_chr, 
                               mut, distinct = F){
    # TODO
    #   - allow for country to define haplotype_consensus
    
    # Prints for checking output
    print(dim(brca_data)) # 8x67
    print(dim(haplotypes_brca)) # 15679x12468
    print(dim(chr_coords)) # 12467x3
    print(mut)
    
    # Make brca_data a global variable (for testing and debugging)
    brca_data_temp <<- brca_data
    brca_data <<- brca_data_temp
    
    
    matched <<- extractSamples(brca_data, haplotypes_brca, distinct)
    
    chr_coords2 = filter(chr_coords, chr_coords$Chr_numeric == brca_chr)
    matched <- select(matched, na.omit(fmatch(colnames(matched[2:ncol(matched)]), chr_coords2$SNP)))
    # matched <- matched[, na.omit(fmatch(colnames(matched[2:ncol(matched)]), chr_coords2$SNP))]
    
    haplotypes <<- findConsensus(matched)
    # haplotypes <<- findConsensus(filter(matched, SNP %in% subset(brca_data, Country == "SPAIN")$Onc_ID))
    
    breaks <<- findHaplotypeBreaks(matched, haplotypes)
    
    if (length(breaks) == 0){
        print(paste("No haplotype breaks found in families for mutation:", mut))
        return()
    }
    
    chr_pos <<- mapSNPs(breaks, chr_coords, brca_middle, brca_start, brca_stop, brca_chr)
    #chr_pos <- mapSNPs(breaks, chr_coords, brca1_middle, brca1_start, brca1_stop)
    #chr_pos <- mapSNPs()
    save(chr_pos, file = paste0("chr_pos-", brca_name, "-", paste(mut, collapse = ","), ".RData"))
    
    plot_haplotype(chr_pos, brca_chr, mut)
    
    findNearestBreaks(chr_pos)
}
